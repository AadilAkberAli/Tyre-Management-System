        private void bunifuThinButton22_Click(object sender, EventArgs e)
        {
            con.Open();
            SqlCommand cnd = new SqlCommand("Select name,streetaddress,phone,zip from Table5", con);
            SqlDataReader da = cnd.ExecuteReader();
            while (da.Read())
            {
                namecustomer.Text = da.GetValue(0).ToString();
                Address.Text = da.GetValue(1).ToString();
                Phone.Text = da.GetValue(2).ToString();
                invoicenumber.Text = da.GetValue(3).ToString();
            }
            con.Close();
            Dateofissue.Text = DateTime.Now.ToLongDateString();
            disp_data();
           
        }
